# --- Bonsai Ternary Fork (PrismML-Eng, branch prism; Vulkan-Q2_0 ist drin) ---
git clone https://github.com/PrismML-Eng/llama.cpp.git _tmp_2b_llama
Push-Location .\_tmp_2b_llama
git checkout prism
# Upstream-Tags holen und die b-Nummer der MAINLINE-Basis ermitteln
# (PrismML taggt selbst als prism-bXXXX; wir wollen die echte Basis).
git remote add upstream https://github.com/ggml-org/llama.cpp.git 2>$null
git fetch upstream --tags 2>$null
$base = git merge-base HEAD upstream/master 2>$null
if (-not $base) { $base = "HEAD" }
$desc = git describe --tags --abbrev=0 --match "b[0-9]*" $base 2>$null
$ver  = [regex]::Match($desc, 'b\d+').Value
if (-not $ver) { $ver = "bUNKNOWN" }
Pop-Location
$dir = "2b_${ver}_llama.cpp"
if (Test-Path .\$dir) { Remove-Item .\$dir -Recurse -Force }
Rename-Item .\_tmp_2b_llama $dir
Write-Host "==> Build-Verzeichnis: $dir  (Fork-Basis auf mainline $ver)"

# --- UI: aus Source bauen wenn UI-Quellen vorhanden, sonst Prebuilt von HF ---
# Vor b9174 (16.05.2026) lag die UI unter tools/server/webui/,
# ab b9174 unter tools/ui/. Aeltere Forks (z.B. PrismML prism-Branch)
# fuehren ggf. gar keine UI-Quellen mit -> dann Prebuilt von HF holen.
$uiSrc = $null
foreach ($cand in ("tools/ui", "tools/server/webui")) {
    if (Test-Path .\$dir\$cand\package.json) { $uiSrc = $cand; break }
}
if ($uiSrc) {
    Write-Host "==> UI aus Source bauen ($dir\$uiSrc)"
    Push-Location .\$dir\$uiSrc
    if (Test-Path package-lock.json) { npm ci } else { npm install }
    npm run build
    Pop-Location
    $uiPrebuilt = "OFF"
} else {
    Write-Host "==> keine UI-Quellen in $dir — verwende Prebuilt-UI von HF"
    $uiPrebuilt = "ON"
}

# LLAMA_OPENSSL=OFF: der prism-Branch basiert auf einer aelteren Mainline
# (vor b9174) und vendort noch cpp-httplib 0.40.0. Dessen httplib.cpp:12574
# weist X509_get_subject_name(cert) einer non-const Variablen zu; ab
# OpenSSL 3.2 liefert diese Funktion aber const X509_NAME* -> C2440
# ("Conversion loses qualifiers"). Fuer die lokale Nutzung auf 127.0.0.1
# ist HTTPS ohnehin unnoetig, also SSL im Build deaktivieren.
# Neuere Forks (tq_/d_/mainline b9840+) haben ein gefixtes cpp-httplib
# und brauchen diesen Schalter nicht.

# CMAKE_POLICY_DEFAULT_CMP0194=OLD + CMAKE_ASM_COMPILER=cl umgehen den
# ASM-Compiler-Fehler (llama.cpp Issue #22100) bei neueren CMake-Versionen.
cmake -S .\$dir -B .\$dir\build `
  -G "Visual Studio 18 2026" `
  -A x64 `
  -DCMAKE_BUILD_TYPE=Release `
  -DCMAKE_POLICY_DEFAULT_CMP0194=OLD `
  -DCMAKE_ASM_COMPILER=cl `
  -DLLAMA_OPENSSL=OFF `
  -DGGML_VULKAN=ON `
  -DGGML_NATIVE=OFF `
  -DGGML_AVX2=ON `
  -DGGML_FMA=ON `
  -DGGML_F16C=ON `
  -DBUILD_SHARED_LIBS=OFF `
  -DLLAMA_BUILD_SERVER=ON `
  -DLLAMA_BUILD_UI=ON `
  -DLLAMA_USE_PREBUILT_UI=$uiPrebuilt `
  -DLLAMA_CURL=OFF `
  -DGGML_CCACHE=OFF `
  -DGGML_VULKAN_CHECK_RESULTS=OFF `
  -DCMAKE_PREFIX_PATH="H:/LAB/ai-local/SPIRV-Headers/install"
cmake --build .\$dir\build --config Release --parallel 24
